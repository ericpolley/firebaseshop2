This website is a shop using firebase and reactjs.

Installs:

npx create-react-app@latest ./

npm install -D tailwindcss postcss autoprefixer

npx tailwindcss init -p

content: [
"./src/**/*.{js,jsx,ts,tsx}",
],

@tailwind base;
@tailwind components;
@tailwind utilities;

npm install

npm run start
